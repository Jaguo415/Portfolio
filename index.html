-- Cohort month and month offset
WITH events AS (
  SELECT user_id, date_trunc('month', signup_dt) AS cohort,
         date_trunc('month', activity_dt) AS period
  FROM fact_activity
)
SELECT cohort, period, count(distinct user_id)
FROM events
GROUP BY 1,2;
